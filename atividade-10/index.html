Objeto Funcionário - 3 Formas de Criar

1. Sintaxe Literal de Objeto

const funcionario1 = {
    matricula: "F12345",
    nome: "Ana Pereira",
    funcao: "Gerente de Vendas"
};

2. Função Construtora

function Funcionario(matricula, nome, funcao) {
    this.matricula = matricula;
    this.nome = nome;
    this.funcao = funcao;
}
const funcionario2 = new Funcionario("F67890", "Carlos Santos", "Desenvolvedor");

3. Object.create()

const funcionarioProto = {
    exibirDados: function() {
        return `Matrícula: ${this.matricula}, Nome: ${this.nome}, Função: ${this.funcao}`;
    }
};
const funcionario3 = Object.create(funcionarioProto);
funcionario3.matricula = "F54321";
funcionario3.nome = "Mariana Oliveira";
funcionario3.funcao = "Designer";
